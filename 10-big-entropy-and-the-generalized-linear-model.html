      def sim_p(i, G=1.4):
    x123 = dist.Uniform().sample(PRNGKey(i), (3,))
    x4 = (G * np.sum(x123, keepdims=True) - x123[1] - x123[2]) / (2 - G)
    z = np.sum(np.concatenate([x123, x4]))
    p = np.concatenate([x123, x4]) / z
    return {"H": -np.sum(p * np.log(p)), "p": p}
